P1 # P2 = 
 + P1 _als P2=O_
 + P2 _als P1=O_
 + O  _als x1=x2 en y1!=y2_
 + O  _als x1=x2 en y1=y2=0_
 + P3 _overige gevallen

x3 = -x1 -x2 -b *a^-1 +m^2 *a^-1  
y3 = -y1 +m (x1 -x2)  
m = 
 + (y2 -y1) (x2 -x1)^-1 _als P1!=P2_
 + (3 *a *x1^2 +2 *b *x1 +c) *2^-1 *y1^-1 _als P1=P2_
